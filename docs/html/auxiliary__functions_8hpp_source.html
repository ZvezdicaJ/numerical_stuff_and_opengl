66 inline __m128 load_vertex(float *value) {
67  // load the x and y element of the float3 vector using a 64 bit load
68  // and set the upper 64 bits to zero (00YX)
69  __m128 xy = _mm_loadl_pi(_mm_setzero_ps(), (__m64 *)value);
70  // now load the z element using a 32 bit load (000Z)
71  __m128 z = _mm_load_ss(value + 2);
72  // finally, combine both by moving the z component to the high part
73  // of the register, while keeping x and y components in the low part
74  return _mm_movelh_ps(xy, z); // (0ZYX)
