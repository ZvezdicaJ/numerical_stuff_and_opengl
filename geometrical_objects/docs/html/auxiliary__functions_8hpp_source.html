44 inline __m128 load_vertex(float *value) {
45  // load the x and y element of the float3 vector using a 64 bit load
46  // and set the upper 64 bits to zero (00YX)
47  __m128 xy = _mm_loadl_pi(_mm_setzero_ps(), (__m64 *)value);
48  // now load the z element using a 32 bit load (000Z)
49  __m128 z = _mm_load_ss(value + 2);
50  // finally, combine both by moving the z component to the high part
51  // of the register, while keeping x and y components in the low part
52  return _mm_movelh_ps(xy, z); // (0ZYX)
53 }
